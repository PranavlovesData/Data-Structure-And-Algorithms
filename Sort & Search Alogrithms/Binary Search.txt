class Solution:    
    def binarysearch(self, arr, n, k):
        arr.sort()
        start=0
        end=n-1
        while start<=end:
            mid=(start+end)//2
            if arr[mid]>k:
                end=mid-1
            elif arr[mid]<k:
                start=mid +1
            else:
                return mid
        else:
            return -1
